118  template<class ...Args>
119  bool emplace(Args&&... record_args) {
120  size_t const current_write = write_idx.load(std::memory_order_relaxed);
121  size_t next_record = current_write + 1;
122  if (next_record == ring_size) {
123  next_record = 0;
124  }
125  if (next_record != read_idx.load(std::memory_order_acquire)) {
126  new (std::addressof(records[current_write])) T(std::forward<Args>(record_args)...);
127  write_idx.store(next_record, std::memory_order_release);
128  return true;
129  }
130 
131  // queue is full
132  return false;
133  }

141  bool poll(T& record) {
142  size_t const current_read = read_idx.load(std::memory_order_relaxed);
143  if (current_read == write_idx.load(std::memory_order_acquire)) {
144  // queue is empty
145  return false;
146  }
147 
148  size_t next_record = current_read + 1;
149  if (next_record == ring_size) {
150  next_record = 0;
151  }
152  record = std::move(records[current_read]);
153  records[current_read].~T();
154  read_idx.store(next_record, std::memory_order_release);
155  return true;
156  }
