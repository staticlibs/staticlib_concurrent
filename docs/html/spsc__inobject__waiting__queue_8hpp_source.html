120  bool take(T& record, std::chrono::milliseconds timeout = std::chrono::milliseconds(0)) {
121  bool res = queue.poll(record);
122  if (res) {
123  return true;
124  }
125  std::unique_lock<std::mutex> guard{mutex};
126  auto predicate = [this] {
127  return this->unblocked || !this->queue.empty();
128  };
129  if (std::chrono::milliseconds(0) == timeout) {
130  empty_cv.wait(guard, predicate);
131  } else {
132  empty_cv.wait_for(guard, timeout, predicate);
133  }
134  return queue.poll(record);
