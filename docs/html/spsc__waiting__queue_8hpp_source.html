119  bool take(T& record, std::chrono::milliseconds timeout = std::chrono::milliseconds(0)) {
120  bool res = queue.poll(record);
121  if (res) {
122  return true;
123  }
124  std::unique_lock<std::mutex> guard{mutex};
125  auto predicate = [this] {
126  return this->unblocked || !this->queue.empty();
127  };
128  if (std::chrono::milliseconds(0) == timeout) {
129  empty_cv.wait(guard, predicate);
130  } else {
131  empty_cv.wait_for(guard, timeout, predicate);
132  }
133  return queue.poll(record);
134  }
