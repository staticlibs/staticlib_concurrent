107  template<class ...Args>
108  bool emplace(Args&&... record_args) {
109  size_t const current_write = write_idx.load(std::memory_order_relaxed);
110  size_t next_record = current_write + 1;
111  if (next_record == ring_size) {
112  next_record = 0;
113  }
114  if (next_record != read_idx.load(std::memory_order_acquire)) {
115  new (std::addressof(records[current_write])) T(std::forward<Args>(record_args)...);
116  write_idx.store(next_record, std::memory_order_release);
117  return true;
118  }

120  // queue is full

130  bool poll(T& record) {
131  size_t const current_read = read_idx.load(std::memory_order_relaxed);
132  if (current_read == write_idx.load(std::memory_order_acquire)) {
133  // queue is empty
134  return false;
135  }
136 
137  size_t next_record = current_read + 1;
138  if (next_record == ring_size) {
139  next_record = 0;
140  }
141  record = std::move(records[current_read]);
142  records[current_read].~T();
143  read_idx.store(next_record, std::memory_order_release);
144  return true;
